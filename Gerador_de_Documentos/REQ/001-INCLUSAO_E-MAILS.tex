\section{DOS PEDIDOS}

\newcommand{\tipoRequerimento}{Solicitação de Inclusão de E-mails}


\begin{enumerate}[resume]
  \item Diante do exposto, requer-se, a contar desta data, a inclusão do endereço de e-mail \textbf{\EmailEmpresa} na lista oficial de distribuição de comunicados deste Município.
  \label{item:email_empresa}
  \item Para fins de transparência e adequada ciência do Município, requer-se que toda e qualquer comunicação futura relativa a Termo de Ocorrência e Inspeção (TOI), Quadro de Iluminação Pública (QIP), recontagem do parque de iluminação pública, arrecadação da CIP, faturas de energia e demais assuntos correlatos seja encaminhada também com cópia para o e-mail indicado no item anterior, sem prejuízo dos demais endereços eletrônicos já cadastrados junto à Concessionária.
  \item Solicita-se o envio, para o e-mail \EmailEmpresa, dos últimos e-mails encaminhados ao Município contendo informações sobre o Quadro de Iluminação Pública (QIP), os valores arrecadados pela Contribuição de Iluminação Pública (CIP) e as Faturas de Energia, referentes aos últimos 3 meses.
\end{enumerate}
